import random # Irregular verbs dictionary irregular_verbs = { "eat": ("ate", "eaten"), "go": ("went", "gone"), "see": ("saw", "seen"), "write": ("wrote", "written"), "take": ("took", "taken"), "play": ("played", "played"), # regular verb example "walk": ("walked", "walked") # regular verb example } # All 12 tenses templates tenses = [ ("Simple Present", "{subject} {verb}"), ("Present Continuous", "{subject} am/is/are {verb}ing"), ("Present Perfect", "{subject} have/has {past_participle}"), ("Present Perfect Continuous", "{subject} have/has been {verb}ing"), ("Simple Past", "{subject} {past}"), ("Past Continuous", "{subject} was/were {verb}ing"), ("Past Perfect", "{subject} had {past_participle}"), ("Past Perfect Continuous", "{subject} had been {verb}ing"), ("Simple Future", "{subject} will {verb}"), ("Future Continuous", "{subject} will be {verb}ing"), ("Future Perfect", "{subject} will have {past_participle}"), ("Future Perfect Continuous", "{subject} will have been {verb}ing") ] # Function to pick past/past participle def get_past_forms(verb): return irregular_verbs.get(verb, (verb + "ed", verb + "ed")) # Function to generate a random quiz question def generate_question(): subject = random.choice(["I", "You", "He", "She", "They"]) verb = random.choice(list(irregular_verbs.keys())) past, past_participle = get_past_forms(verb) tense_name, template = random.choice(tenses) sentence = template.format(subject=subject, verb=verb, past=past, past_participle=past_participle) return tense_name, sentence, subject, verb # Quiz loop score = 0 num_questions = 5 print("=== Welcome to the Tense Quiz ===") print("Fill in the blanks with the correct tense.\n") for i in range(num_questions): tense_name, sentence, subject, verb = generate_question() print(f"Question {i+1}: ({tense_name})") # Create a blank blank_sentence = sentence.replace(verb, "_____") user_answer = input(blank_sentence + " ") # Check answer if user_answer.strip().lower() == verb.lower(): print("✅ Correct!\n") score += 1 else: print(f"❌ Wrong! Correct answer: {verb}\n") print(f"Your final score: {score}/{num_questions}") print("Quiz complete! 🎉")
